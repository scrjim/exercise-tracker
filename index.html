<script>
// ── CONFIG ───────────────────────────────────────────────────
const CLIENT_ID = '688992356709-l1qkvk7tohg4ibs001cec1g0ddll3dl6.apps.googleusercontent.com';
const SCOPES = 'https://www.googleapis.com/auth/spreadsheets https://www.googleapis.com/auth/drive.file';
const SHEET_ID_KEY = 'exercise_sheet_id';
const SHEET_NAME = 'ExerciseData';

// ── EXERCISE DEFINITIONS ─────────────────────────────────────
const WEIGHTED = ['Squat','Push up','Chin-up','SLDL','DB shoulder press','Bent over row'];
const TIMED = ['Plank','Wall sit','Dead hang'];
const ALL_EX = [...WEIGHTED, ...TIMED];
const DOW = ['Sun','Mon','Tue','Wed','Thu','Fri','Sat'];
const MONTHS = ['January','February','March','April','May','June','July','August','September','October','November','December'];

// ── STATE ────────────────────────────────────────────────────
let accessToken = null;
let sheetId = localStorage.getItem(SHEET_ID_KEY);
let data = {};
let view = 'month';
let cursor = new Date(); cursor.setHours(0,0,0,0);
let summaryEx = WEIGHTED[0];
let saveTimer = null;
let pendingSaves = {};

// ── SCREENS ──────────────────────────────────────────────────
function showScreen(id) {
document.querySelectorAll('.screen').forEach(s => s.classList.remove('active'));
document.getElementById('screen-' + id).classList.add('active');
}

// ── AUTH ─────────────────────────────────────────────────────
function signIn() {
const client = google.accounts.oauth2.initTokenClient({
client_id: CLIENT_ID,
scope: SCOPES,
callback: async (resp) => {
if (resp.error) { showToast('Sign in failed'); return; }
accessToken = resp.access_token;
await afterSignIn();
}
});
client.requestAccessToken();
}

async function afterSignIn() {
if (sheetId) {
showScreen('loading');
const ok = await loadSheetData();
if (ok) { showScreen('app'); render(); }
else { sheetId = null; localStorage.removeItem(SHEET_ID_KEY); showScreen('setup'); }
} else {
showScreen('setup');
}
}

function signOut() {
if (accessToken) google.accounts.oauth2.revoke(accessToken);
accessToken = null;
data = {};
showScreen('gate');
}

// ── SHEET SETUP ──────────────────────────────────────────────
async function createNewSheet() {
showScreen('loading');
try {
const res = await gapi('https://sheets.googleapis.com/v4/spreadsheets', 'POST', {
properties: { title: "Jim's Exercise Tracker" },
sheets: [{ properties: { title: SHEET_NAME } }]
});
sheetId = res.spreadsheetId;
localStorage.setItem(SHEET_ID_KEY, sheetId);
await initSheetHeaders();
data = {};
showScreen('app');
render();
showToast('Sheet created in your Google Drive!');
} catch(e) {
showToast('Could not create sheet — try again');
showScreen('setup');
}
}

async function useExistingSheet() {
const id = document.getElementById('existing-sheet-id').value.trim();
if (!id) { showToast('Please paste a Sheet ID'); return; }
sheetId = id;
localStorage.setItem(SHEET_ID_KEY, sheetId);
showScreen('loading');
const ok = await loadSheetData();
if (ok) { showScreen('app'); render(); }
else { showToast('Could not access that sheet'); showScreen('setup'); }
}

async function initSheetHeaders() {
const headers = buildHeaders();
await gapi(`https://sheets.googleapis.com/v4/spreadsheets/${sheetId}/values/${SHEET_NAME}!A1:append?valueInputOption=RAW`, 'POST', {
values: [headers]
});
}

function buildHeaders() {
const h = ['date'];
WEIGHTED.forEach(ex => {
for (let s=1; s<=3; s++) { h.push(`${ex}|s${s}|reps`); h.push(`${ex}|s${s}|weight`); }
});
TIMED.forEach(ex => {
for (let s=1; s<=4; s++) { h.push(`${ex}|s${s}|time`); }
});
return h;
}

// ── SHEET DATA ───────────────────────────────────────────────
async function loadSheetData() {
try {
const res = await gapi(`https://sheets.googleapis.com/v4/spreadsheets/${sheetId}/values/${SHEET_NAME}`, 'GET');
const rows = res.values || [];
if (rows.length < 1) { await initSheetHeaders(); return true; }
const headers = rows[0];
data = {};
for (let i=1; i<rows.length; i++) {
const row = rows[i];
const dateKey = row[0];
if (!dateKey) continue;
data[dateKey] = {};
headers.forEach((h, idx) => {
if (h !== 'date' && row[idx]) data[dateKey][h] = row[idx];
});
}
setSyncDot('synced'); setSyncLabel('synced');
return true;
} catch(e) {
return false;
}
}

async function saveDay(dayKey) {
if (!sheetId || !accessToken) return;
setSyncDot('syncing'); setSyncLabel('saving…');
try {
const res = await gapi(`https://sheets.googleapis.com/v4/spreadsheets/${sheetId}/values/${SHEET_NAME}`, 'GET');
const rows = res.values || [];
const headers = rows.length > 0 ? rows[0] : buildHeaders();
if (rows.length === 0) await initSheetHeaders();

const dd = data[dayKey] || {};
const rowData = headers.map(h => h === 'date' ? dayKey : (dd[h] || ''));

let rowIndex = -1;
for (let i=1; i<rows.length; i++) {
if (rows[i][0] === dayKey) { rowIndex = i+1; break; }
}

if (rowIndex > 0) {
await gapi(`https://sheets.googleapis.com/v4/spreadsheets/${sheetId}/values/${SHEET_NAME}!A${rowIndex}?valueInputOption=RAW`, 'PUT', { values: [rowData] });
} else {
await gapi(`https://sheets.googleapis.com/v4/spreadsheets/${sheetId}/values/${SHEET_NAME}!A1:append?valueInputOption=RAW`, 'POST', { values: [rowData] });
}
setSyncDot('synced'); setSyncLabel('saved');
} catch(e) {
setSyncDot('error'); setSyncLabel('error');
showToast('Save failed — check connection');
}
}

// ── API HELPER ───────────────────────────────────────────────
async function gapi(url, method, body) {
const opts = {
method,
headers: { 'Authorization': `Bearer ${accessToken}`, 'Content-Type': 'application/json' }
};
if (body) opts.body = JSON.stringify(body);
const res = await fetch(url, opts);
if (!res.ok) throw new Error(await res.text());
return res.json();
}

// ── DATA HELPERS ─────────────────────────────────────────────
function dayKey(d) { return d.getFullYear()+'-'+(d.getMonth()+1)+'-'+d.getDate(); }
function getDayData(d) { return data[dayKey(d)] || {}; }
function hasData(d) { const dd=getDayData(d); return Object.values(dd).some(v=>v!==''&&v!=null); }

function saveField(input) {
const k=input.dataset.k, f=input.dataset.f, v=input.value;
if (!data[k]) data[k]={};
data[k][f]=v;
scheduleSave(k);
}

// ── VIEWS ────────────────────────────────────────────────────
const today = new Date(); today.setHours(0,0,0,0);
function isToday(d) { return d.toDateString()===today.toDateString(); }

function setView(v) {
view=v;
document.querySelectorAll('.view-tab').forEach((b,i)=>b.classList.toggle('active',['month','week','day'][i]===v));
if(v==='day') openDayModal(cursor);

function navPrev() {
if(view==='month') cursor=new Date(cursor.getFullYear(),cursor.getMonth()-1,1);
else if(view==='week') cursor=new Date(cursor.getTime()-7*86400000);
else { cursor=new Date(cursor.getTime()-86400000); openDayModal(cursor); }
render();
}
function navNext() {
if(view==='month') cursor=new Date(cursor.getFullYear(),cursor.getMonth()+1,1);
else if(view==='week') cursor=new Date(cursor.getTime()+7*86400000);
else { cursor=new Date(cursor.getTime()+86400000); openDayModal(cursor); }
render();
}

function render() {
if(view==='month') renderMonth();
else if(view==='week') renderWeek();
else renderDayNav();
renderSummary();
}

function renderMonth() {
document.getElementById('nav-label').innerHTML=`${MONTHS[cursor.getMonth()]} <span>${cursor.getFullYear()}</span>`;
const first=new Date(cursor.getFullYear(),cursor.getMonth(),1);
const last=new Date(cursor.getFullYear(),cursor.getMonth()+1,0);
let html='<div class="cal-grid">';
DOW.forEach(d=>html+=`<div class="cal-dow">${d}</div>`);
for(let i=0;i<first.getDay();i++) html+=calCell(new Date(first.getTime()-(first.getDay()-i)*86400000),true);
for(let d=1;d<=last.getDate();d++) html+=calCell(new Date(cursor.getFullYear(),cursor.getMonth(),d),false);
const filled=(first.getDay()+last.getDate())%7;
if(filled>0) for(let i=1;i<=7-filled;i++) html+=calCell(new Date(cursor.getFullYear(),cursor.getMonth()+1,i),true);
html+='</div>';
document.getElementById('main-view').innerHTML=html;
}

function calCell(d,other) {
const dd=getDayData(d);
let dots=ALL_EX.map(ex=>{
const on=Object.keys(dd).some(f=>f.startsWith(ex)&&dd[f]!=='');
return `<div class="cal-dot ${on?'on':''}"></div>`;
}).join('');
const cls=['cal-cell',other?'other-month':'',isToday(d)?'today':'',hasData(d)?'has-data':''].filter(Boolean).join(' ');
return `<div class="${cls}" onclick="openDayModal(new Date(${d.getTime()}))">
<div class="cal-dn">${d.getDate()}</div>
<div class="cal-dots">${dots}</div>
</div>`;
}

function renderWeek() {
const ws=weekStart(cursor);
const we=new Date(ws.getTime()+6*86400000);
const ml=MONTHS[ws.getMonth()].slice(0,3);
const sfx=we.getMonth()!==ws.getMonth()?` – ${MONTHS[we.getMonth()].slice(0,3)} ${we.getDate()}`:` – ${we.getDate()}`;
document.getElementById('nav-label').innerHTML=`${ml} ${ws.getDate()}${sfx} <span>${we.getFullYear()}</span>`;
let html='<div class="week-rows">';
for(let i=0;i<7;i++){
const d=new Date(ws.getTime()+i*86400000);
const isTod=isToday(d);
const dd=getDayData(d);
const done=ALL_EX.filter(ex=>Object.keys(dd).some(f=>f.startsWith(ex)&&dd[f]!==''));
const label=done.length?done.slice(0,5).join(', ')+(done.length>5?'…':''):'— tap to log';
html+=`<div class="week-row">
<div class="week-label ${isTod?'today':''}">
<div class="wd">${DOW[d.getDay()]}</div>
<div class="wdate">${d.getDate()}</div>
</div>
<div class="week-cell ${isTod?'today-cell':''}" onclick="openDayModal(new Date(${d.getTime()}))">
<div class="week-cell-content ${done.length?'has':''}">${label}</div>
</div>
</div>`;
}
html+='</div>';
document.getElementById('main-view').innerHTML=html;
}

document.getElementById('main-view').innerHTML='';
}

// ── SUMMARY ──────────────────────────────────────────────────
function weekStart(d){ const ws=new Date(d); ws.setDate(ws.getDate()-ws.getDay()); ws.setHours(0,0,0,0); return ws; }
function weekDays(d){ const ws=weekStart(d); return Array.from({length:7},(_,i)=>new Date(ws.getTime()+i*86400000)); }

function renderSummary(){
document.getElementById('ex-pills').innerHTML=ALL_EX.map(ex=>
`<button class="ex-pill ${ex===summaryEx?'active':''}" onclick="setSummaryEx('${ex}')">${ex}</button>`
).join('');
const days=weekDays(view==='month'?today:cursor);
const ex=summaryEx, isTimed=TIMED.includes(ex);
let cards='';
if(!isTimed){
let totalReps=0,totalVol=0,bestW=0;
days.forEach(d=>{
const dd=getDayData(d);
for(let s=1;s<=3;s++){
const r=parseFloat(dd[`${ex}|s${s}|reps`])||0;
const w=parseFloat(dd[`${ex}|s${s}|weight`])||0;
totalReps+=r; totalVol+=r*w; if(w>bestW) bestW=w;
}
});
cards+=sCard('Total reps',totalReps,'this week');
cards+=sCard('Volume',Math.round(totalVol),'kg this week');
cards+=sCard('Best weight',bestW,'kg lifted');
} else {
let totalTime=0,bestSet=0,totalSets=0;
days.forEach(d=>{
const dd=getDayData(d);
for(let s=1;s<=4;s++){
const t=parseFloat(dd[`${ex}|s${s}|time`])||0;
if(t>0){totalTime+=t;totalSets++;if(t>bestSet)bestSet=t;}
}
});
cards+=sCard('Total time',totalTime,'sec this week');
cards+=sCard('Best set',bestSet,'seconds');
cards+=sCard('Sets logged',totalSets,'this week');
}
document.getElementById('stat-cards').innerHTML=cards;
}

function sCard(label,val,unit){
return `<div class="stat-card"><div class="stat-card-label">${label}</div><div class="stat-card-val">${val}</div><div class="stat-card-unit">${unit}</div></div>`;
}
function setSummaryEx(ex){ summaryEx=ex; renderSummary(); }
